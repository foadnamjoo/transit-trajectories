script.js 
// System build - compact version (D3 v7)
const parseDate = d3.timeParse("%Y-%m-%d");
let STATE = { rows: [], metric: "ridership" };
const REFS = {};

document.addEventListener("DOMContentLoaded", () => {
  setup();
  loadData();
});

function setup(){
  d3.select("#dataset").on("change", loadData);
  d3.select("#metric").on("change", () => { STATE.metric = d3.select("#metric").property("value"); update(); });

  ["Histogram-div","Linechart-div","StackedArea-div","Scatterplot-div"].forEach((id) => {
    const svg = d3.select("#"+id).append("svg").attr("viewBox","0 0 500 260");
    const M = {l:50,r:16,t:20,b:30}, W = 500 - M.l - M.r, H = 260 - M.t - M.b;
    const gPlot = svg.append("g").attr("class","plot").attr("transform",`translate(${M.l},${M.t})`);
    const gX = svg.append("g").attr("class","x-axis").attr("transform",`translate(${M.l},${M.t+H})`);
    const gY = svg.append("g").attr("class","y-axis").attr("transform",`translate(${M.l},${M.t})`);
    const xL = svg.append("text").attr("class","x-label").attr("text-anchor","middle").attr("x",M.l+W/2).attr("y",254);
    const yL = svg.append("text").attr("class","y-label").attr("text-anchor","middle").attr("transform",`translate(14, ${M.t+H/2}) rotate(-90)`);
    const legend = svg.append("g").attr("class","legend").attr("transform","translate(380,24)");
    REFS[id] = {svg,gPlot,gX,gY,xL,yL,legend,W,H,M};
  });
}

function loadData(){
  const file = d3.select("#dataset").property("value");
  d3.csv("data/"+file).then(raw => {
    STATE.rows = raw.map(d => ({
      date: parseDate(d.date),
      group: d.group,
      ridership: +d.ridership,
      on_time_pct: +d.on_time_pct,
      day_type: d.day_type
    })).filter(d => d.date);
    STATE.rows.sort((a,b)=>a.date-b.date);
    update();
  }).catch(err => console.error(err));
}

const dayTypeColor = d3.scaleOrdinal().domain(["Weekday","Weekend","Holiday"]).range(["#e41a1c","#377eb8","#4daf4a"]);

function update(){
  if(!STATE.rows.length) return;
  drawHist(); drawLine(); drawStack(); drawScatter();
}

function drawHist(){
  const {gPlot,gX,gY,xL,yL,W,H} = REFS["Histogram-div"];
  const m = STATE.metric, vals = STATE.rows.map(d=>d[m]);
  const x = d3.scaleLinear().domain(d3.extent(vals)).nice().range([0,W]);
  const bins = d3.bin().domain(x.domain()).thresholds(24)(vals);
  const y = d3.scaleLinear().domain([0, d3.max(bins, d=>d.length)||1]).nice().range([H,0]);
  const bars = gPlot.selectAll("rect.bar").data(bins, d=>`${d.x0}-${d.x1}`);
  bars.join(
    e=>e.append("rect").attr("class","bar").attr("x",d=>x(d.x0)+1).attr("width",d=>x(d.x1)-x(d.x0)-1)
      .attr("y",H).attr("height",0).transition().duration(500).attr("y",d=>y(d.length)).attr("height",d=>H-y(d.length)),
    u=>u.transition().duration(500).attr("x",d=>x(d.x0)+1).attr("width",d=>x(d.x1)-x(d.x0)-1).attr("y",d=>y(d.length)).attr("height",d=>H-y(d.length))
  );
  gX.call(d3.axisBottom(x).ticks(6));
  gY.call(d3.axisLeft(y).ticks(5));
  xL.text(m); yL.text("Count");
}

function drawLine(){
  const {gPlot,gX,gY,xL,yL,W,H} = REFS["Linechart-div"];
  const m = STATE.metric;
  const x = d3.scaleTime().domain(d3.extent(STATE.rows,d=>d.date)).range([0,W]);
  const y = d3.scaleLinear().domain(d3.extent(STATE.rows,d=>d[m])).nice().range([H,0]);
  const line = d3.line().x(d=>x(d.date)).y(d=>y(d[m]));
  gPlot.selectAll("path.line").data([STATE.rows]).join(
    e=>e.append("path").attr("class","line").attr("d",line).attr("opacity",0).transition().duration(500).attr("opacity",1),
    u=>u.transition().duration(500).attr("d",line)
  );
  const pts = gPlot.selectAll("circle.pt").data(STATE.rows, d=>+d.date);
  pts.join(
    e=>e.append("circle").attr("class","pt").attr("r",3).attr("cx",d=>x(d.date)).attr("cy",d=>y(d[m]))
      .attr("fill",d=>dayTypeColor(d.day_type)).attr("opacity",0).transition().duration(300).attr("opacity",0.9),
    u=>u.transition().duration(500).attr("cx",d=>x(d.date)).attr("cy",d=>y(d[m]))
  );
  gX.call(d3.axisBottom(x).ticks(6)); gY.call(d3.axisLeft(y).ticks(5));
  xL.text("Date"); yL.text(m);
}

function drawStack(){
  const {gPlot,gX,gY,xL,yL,legend,W,H} = REFS["StackedArea-div"];
  const table = STATE.rows.map(d=>({date:d.date,ridership:d.ridership,on_time_pct:d.on_time_pct}));
  const x = d3.scaleTime().domain(d3.extent(table,d=>d.date)).range([0,W]);
  const keys=["ridership","on_time_pct"], stack = d3.stack().keys(keys)(table);
  const y = d3.scaleLinear().domain([0,d3.max(stack[stack.length-1],d=>d[1])||1]).nice().range([H,0]);
  const area = d3.area().x(d=>x(d.data.date)).y0(d=>y(d[0])).y1(d=>y(d[1]));
  const col = d3.scaleOrdinal().domain(keys).range(["#8cb3ff","#c2eabd"]);
  gPlot.selectAll("path.layer").data(stack, d=>d.key).join(
    e=>e.append("path").attr("class","layer").attr("fill",d=>col(d.key)).attr("d",area).attr("opacity",0)
      .transition().duration(500).attr("opacity",0.95),
    u=>u.transition().duration(500).attr("d",area)
  );
  gX.call(d3.axisBottom(x).ticks(6)); gY.call(d3.axisLeft(y).ticks(5));
  if (legend.selectAll("g.item").empty()){
    const items = legend.selectAll("g.item").data(keys).enter().append("g").attr("class","item").attr("transform",(d,i)=>`translate(0,${i*18})`);
    items.append("rect").attr("width",12).attr("height",12).attr("fill",d=>col(d));
    items.append("text").attr("x",18).attr("y",10).text(d=>d);
  }
  xL.text("Date"); yL.text("Stacked value");
}

function drawScatter(){
  const {gPlot,gX,gY,xL,yL,legend,W,H} = REFS["Scatterplot-div"];
  const x = d3.scaleLinear().domain(d3.extent(STATE.rows,d=>d.ridership)).nice().range([0,W]);
  const y = d3.scaleLinear().domain(d3.extent(STATE.rows,d=>d.on_time_pct)).nice().range([H,0]);
  const dots = gPlot.selectAll("circle.dot").data(STATE.rows, d=>+d.date);
  dots.join(
    e=>e.append("circle").attr("class","dot").attr("r",4).attr("cx",d=>x(d.ridership)).attr("cy",d=>y(d.on_time_pct))
      .attr("fill",d=>dayTypeColor(d.day_type)).attr("opacity",0).transition().duration(300).attr("opacity",0.85),
    u=>u.transition().duration(500).attr("cx",d=>x(d.ridership)).attr("cy",d=>y(d.on_time_pct))
  );
  gX.call(d3.axisBottom(x).ticks(6));
  gY.call(d3.axisLeft(y).ticks(5));
  if (legend.selectAll("g.item").empty()){
    const cats = dayTypeColor.domain();
    const items = legend.selectAll("g.item").data(cats).enter().append("g").attr("class","item").attr("transform",(d,i)=>`translate(0,${i*18})`);
    items.append("rect").attr("width",12).attr("height",12).attr("fill",d=>dayTypeColor(d));
    items.append("text").attr("x",18).attr("y",10).text(d=>d);
  }
  xL.text("ridership"); yL.text("on_time_pct");
}

// Formatters and month tick helper
const fmtInt = d3.format(",");
const fmtPct = d3.format(".0f");
const monthFmt = d3.timeFormat("%b");

function yTickFormatFor(metric){
  return metric === "on_time_pct" ? (d)=>fmtPct(d)+"%" : (d)=>fmtInt(d);
}

// ColorBrewer Set1 (3-class) for day_type

  .domain(["Weekday","Weekend","Holiday"])
  .range(["#e41a1c","#377eb8","#4daf4a"]);
